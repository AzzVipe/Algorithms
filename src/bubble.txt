#include<stdio.h>

void swap(int *a, int *b) {
	int temp = *a;
	*a = *b;
	*b = temp;
}

void bubble_sort(int array[], int size) {
	for(int i = 0; i < size -1; i++) {
		for(int j = 0; j < size - i - 1; j++) {
			if(array[j] > array[j + 1])
				swap(&array[j], &array[j + 1]);
		}
	}
}

int main() {
	int array[] = {2, 5, 6, 1, 4};

	printf("Array before sorting: ");

	for(int i = 0; i < 5; i++)
		printf("%d ", array[i]);
	
	puts("");

	bubble_sort(array, 5);
	printf("Array after sorting: ");

	for(int i = 0; i < 5; i++)
		printf("%d ", array[i]);

	puts("");

	return 0;
}
